158 template <class EC>
159 int domainError(EC errorCode) {
160  const std::map<EC, const char*>& registry = getErrorCodeRegistry<EC>();
161  auto iter = registry.find(errorCode);
162  if (iter != registry.end()) {
163  return domainErrorCode(getErrorDomain<EC>(), errorCode, iter->second);
164  }
165  std::string msg = fmt::format("<Unknown error code '{}'>", static_cast<int>(errorCode));
166  return domainErrorCode(getErrorDomain<EC>(), errorCode, msg.c_str());
167 }

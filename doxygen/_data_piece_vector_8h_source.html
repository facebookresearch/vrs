36template <typename T>
37class DataPieceVector : public DataPiece {
38 public:
40 explicit DataPieceVector(const string& label)
41 : DataPiece(label, DataPieceType::Vector, DataLayout::kVariableSize) {}
44 explicit DataPieceVector(const MakerBundle& bundle);
45
48 const string& getElementTypeName() const override {
49 return vrs::getTypeName<T>();
50 }
53 size_t getVariableSize() const override {
54 return sizeof(T) * stagedValues_.size();
55 }
60 size_t collectVariableData(int8_t* data, size_t bufferSize) const override {
61 const size_t writtenSize = min(bufferSize, getVariableSize());
62 if (writtenSize > 0) {
63 memcpy(data, stagedValues_.data(), writtenSize);
64 }
65 return writtenSize;
66 }
67
70 const vector<T>& stagedValues() const {
71 return stagedValues_;
72 }
75 vector<T>& stagedValues() {
76 return stagedValues_;
77 }
80 void stage(const vector<T>& values) {
81 stagedValues_ = values;
82 }
85 void stage(vector<T>&& values) {
86 stagedValues_ = std::move(values);
87 }
88
92 void stage(const T* values, size_t count) {
93 stagedValues_.resize(count);
94 if (count > 0) {
95 memcpy(stagedValues_.data(), values, count * sizeof(T));
96 }
97 }
100 template <size_t n>
101 void stage(const T (&arr)[n]) {
102 stage(arr, n);
103 }
104
108 bool get(vector<T>& outValues) const {
109 size_t count = 0;
110 const T* ptr = layout_.getVarData<T>(offset_, count);
111 if (ptr != nullptr && count > 0) {
112 outValues.resize(count);
113 memcpy(outValues.data(), ptr, count * sizeof(T));
114 return true;
115 }
116 outValues = defaultValues_;
117 return false;
118 }
119
122 const vector<T>& getDefault() const {
123 return defaultValues_;
124 }
128 void setDefault(const T* defaultValues, size_t count) {
129 defaultValues_.resize(count);
130 if (count > 0) {
131 memcpy(defaultValues_.data(), defaultValues, count * sizeof(T));
132 }
133 }
136 template <size_t n>
137 void setDefault(const T (&arr)[n]) {
138 setDefault(arr, n);
139 }
142 void setDefault(const vector<T>& values) {
143 defaultValues_ = values;
144 }
147 void setDefault(vector<T>&& values) {
148 defaultValues_ = std::move(values);
149 }
150
153 bool isAvailable() const override {
154 size_t count = 0;
155 return layout_.getVarData<T>(offset_, count) != nullptr;
156 }
157
162 void print(ostream& out, const string& indent) const override;
167 void printCompact(ostream& out, const string& indent) const override;
168
173 bool isSame(const DataPiece* rhs) const override;
174
178 void serialize(JsonWrapper& jsonWrapper, const JsonFormatProfileSpec& profile) override;
179
182 bool stageCurrentValue() override {
183 return get(stagedValues());
184 }
185
189 unique_ptr<DataPiece> clone() const override {
190 auto other = std::make_unique<DataPieceVector<T>>(getLabel());
191 other->tags_ = tags_;
192 other->required_ = required_;
193 other->defaultValues_ = defaultValues_;
194 return other;
195 }
196
197 protected:
198 bool stageFrom(const DataPiece* original) override {
199 const DataPieceVector<T>* originalVect = reinterpret_cast<const DataPieceVector<T>*>(original);
200 return originalVect->get(stagedValues_);
201 }
202
203 private:
204 vector<T> stagedValues_; // values to write to disk
205 vector<T> defaultValues_;
206};

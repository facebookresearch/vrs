265#pragma pack(push, 1) // tells the compiler that the data might not be aligned
266template <class T>
267struct UnalignedValue {
268 T value;
269};
270#pragma pack(pop)

273template <class T>

275 return reinterpret_cast<const UnalignedValue<T>*>(ptr)->value;
276}

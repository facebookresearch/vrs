27 class DiskFileChunk {
28  public:
29  DiskFileChunk() = default;
30  DiskFileChunk(string path, int64_t offset, int64_t size)
31  : path_{std::move(path)}, offset_{offset}, size_{size} {}
32  DiskFileChunk(DiskFileChunk&& other) noexcept {
33  close();
34  file_ = other.file_;
35  path_ = std::move(other.path_);
36  offset_ = other.offset_;
37  size_ = other.size_;
38  other.file_ = nullptr;
39  other.offset_ = 0;
40  other.size_ = 0;
41  }
42  ~DiskFileChunk() {
43  close();
44  }
45  int create(const string& newPath, const map<string, string>& options) {
46  close();
47  file_ = os::fileOpen(newPath, "wb");
48  if (file_ == nullptr) {
49  return errno;
50  }
51  path_ = newPath;
52  offset_ = 0;
53  size_ = 0;
54 #if IS_ANDROID_PLATFORM()
55  const size_t kBufferingSize = 128 * 1024;
56  return setvbuf(file_, nullptr, _IOFBF, kBufferingSize);
57 #else
58  return SUCCESS;
59 #endif
60  }
61  int open(bool readOnly, const map<string, string>& options) {
62  if (file_ != nullptr) {
63  os::fileClose(file_);
64  }
65  file_ = os::fileOpen(path_, readOnly ? "rb" : "rb+");
66  return file_ != nullptr ? SUCCESS : errno;
67  }
68  bool isOpened() const {
69  return file_ != nullptr;
70  }
71  void rewind() const {
72  ::rewind(file_);
73  }
74  int flush() {
75  return ::fflush(file_) != 0 ? errno : SUCCESS;
76  }
77  int tell(int64_t& outFilepos) const {
78  outFilepos = os::fileTell(file_);
79  return outFilepos < 0 ? errno : SUCCESS;
80  }
81  int seek(int64_t pos, int origin) {
82  return os::fileSeek(file_, pos, origin) != 0 ? errno : SUCCESS;
83  }
84  int read(void* ptr, size_t bufferSize, size_t& outReadSize) const {
85  outReadSize = ::fread(ptr, 1, bufferSize, file_);
86  return bufferSize == outReadSize ? SUCCESS : ::ferror(file_) ? errno : DISKFILE_NOT_ENOUGH_DATA;
87  }
88  int write(const void* data, size_t dataSize, size_t& outWrittenSize) const {
89  outWrittenSize = ::fwrite(data, 1, dataSize, file_);
90  return dataSize == outWrittenSize ? SUCCESS
91  : ::ferror(file_) ? errno
92  : DISKFILE_PARTIAL_WRITE_ERROR;
93  }
94  int truncate(int64_t newSize) {
95  if (os::fileSetSize(file_, newSize) == 0) {
96  size_ = newSize;
97  return SUCCESS;
98  }
99  return errno;
100  }
101  int close() {
102  int error = SUCCESS;
103  if (file_ != nullptr) {
104  error = os::fileClose(file_) != 0 ? errno : SUCCESS;
105  file_ = nullptr;
106  }
107  return error;
108  }
109  bool eof() const {
110  return ::feof(file_) != 0;
111  }
112  int64_t getOffset() const {
113  return offset_;
114  }
115  void setOffset(int64_t newOffset) {
116  offset_ = newOffset;
117  }
118  int64_t getSize() const {
119  return size_;
120  }
121  void setSize(int64_t newSize) {
122  size_ = newSize;
123  }
124  const string& getPath() const {
125  return path_;
126  }
127  bool contains(int64_t fileOffset) const {
128  return fileOffset >= offset_ && fileOffset < offset_ + size_;
129  }
130 
131  private:
132  FILE* file_{}; // may be nullptr or not!
133  string path_; // path of this chunk
134  int64_t offset_{}; // offset of this chunk in the file
135  int64_t size_{}; // size of the chunk
136 };

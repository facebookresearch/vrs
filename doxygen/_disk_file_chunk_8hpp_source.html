31class DiskFileChunk {
32 public:
33 DiskFileChunk() = default;
34 DiskFileChunk(string path, int64_t offset, int64_t size)
35 : path_{std::move(path)}, offset_{offset}, size_{size} {}
36 DiskFileChunk(DiskFileChunk&& other) noexcept {
37 close();
38 file_ = other.file_;
39 path_ = std::move(other.path_);
40 offset_ = other.offset_;
41 size_ = other.size_;
42 other.file_ = nullptr;
43 other.offset_ = 0;
44 other.size_ = 0;
45 }
46 ~DiskFileChunk() {
47 close();
48 }
49 int create(const string& newPath, MAYBE_UNUSED const map<string, string>& options) {
50 close();
51 file_ = os::fileOpen(newPath, "wb");
52 if (file_ == nullptr) {
53 return errno;
54 }
55 path_ = newPath;
56 offset_ = 0;
57 size_ = 0;
58 uint64_t bufferSize = 0;
59 if (helpers::getUInt64(options, "io_buffer_size", bufferSize)) {
60 return setvbuf(file_, nullptr, bufferSize == 0 ? _IONBF : _IOFBF, bufferSize);
61 }
62#if IS_ANDROID_PLATFORM()
63 const size_t kBufferingSize = 128 * 1024;
64 return setvbuf(file_, nullptr, _IOFBF, kBufferingSize);
65#else
66 return SUCCESS;
67#endif
68 }
69 int open(bool readOnly, MAYBE_UNUSED const map<string, string>& options) {
70 if (file_ != nullptr) {
71 os::fileClose(file_);
72 }
73 file_ = os::fileOpen(path_, readOnly ? "rb" : "rb+");
74 return file_ != nullptr ? SUCCESS : errno;
75 }
76 bool isOpened() const {
77 return file_ != nullptr;
78 }
79 void rewind() const {
80 ::rewind(file_);
81 }
82 int flush() {
83 return ::fflush(file_) != 0 ? errno : SUCCESS;
84 }
85 int tell(int64_t& outFilepos) const {
86 outFilepos = os::fileTell(file_);
87 return outFilepos < 0 ? errno : SUCCESS;
88 }
89 int seek(int64_t pos, int origin) {
90 return os::fileSeek(file_, pos, origin) != 0 ? errno : SUCCESS;
91 }
92 int read(void* ptr, size_t bufferSize, size_t& outReadSize) const {
93 outReadSize = ::fread(ptr, 1, bufferSize, file_);
94 return bufferSize == outReadSize ? SUCCESS : ::ferror(file_) ? errno : DISKFILE_NOT_ENOUGH_DATA;
95 }
96 int write(const void* data, size_t dataSize, size_t& outWrittenSize) const {
97 outWrittenSize = ::fwrite(data, 1, dataSize, file_);
98 return dataSize == outWrittenSize ? SUCCESS
99 : ::ferror(file_) ? errno
100 : DISKFILE_PARTIAL_WRITE_ERROR;
101 }
102 int truncate(int64_t newSize) {
103 if (os::fileSetSize(file_, newSize) == 0) {
104 size_ = newSize;
105 return SUCCESS;
106 }
107 return errno;
108 }
109 int close() {
110 int error = SUCCESS;
111 if (file_ != nullptr) {
112 error = os::fileClose(file_) != 0 ? errno : SUCCESS;
113 file_ = nullptr;
114 }
115 return error;
116 }
117 bool eof() const {
118 return ::feof(file_) != 0;
119 }
120 int64_t getOffset() const {
121 return offset_;
122 }
123 void setOffset(int64_t newOffset) {
124 offset_ = newOffset;
125 }
126 int64_t getSize() const {
127 return size_;
128 }
129 void setSize(int64_t newSize) {
130 size_ = newSize;
131 }
132 const string& getPath() const {
133 return path_;
134 }
135 bool contains(int64_t fileOffset) const {
136 return fileOffset >= offset_ && fileOffset < offset_ + size_;
137 }
138
139 private:
140 FILE* file_{}; // may be nullptr or not!
141 string path_; // path of this chunk
142 int64_t offset_{}; // offset of this chunk in the file
143 int64_t size_{}; // size of the chunk
144};
145
146} // namespace vrs

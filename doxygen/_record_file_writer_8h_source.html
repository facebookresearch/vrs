267  bool operator<(const SortRecord& rhs) const {
268  return this->record->getTimestamp() < rhs.record->getTimestamp() ||
269  (this->record->getTimestamp() <= rhs.record->getTimestamp() &&
270  (this->streamId < rhs.streamId ||
271  // Records have a unique creation order within a particular device
272  (this->streamId == rhs.streamId &&
273  this->record->getCreationOrder() < rhs.record->getCreationOrder())));
274  }

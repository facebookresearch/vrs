462  if (mappedDataLayout_ != nullptr) {
463  return mappedDataLayout_->getVarData<T>(varPieceIndex, outCount);
464  }
465  if (varPieceIndex < varSizePieces_.size()) {
466  const IndexEntry& indexEntry = getVarSizeIndex()[varPieceIndex];
467  if (indexEntry.getOffset() + indexEntry.getLength() <= varData_.size()) {
468  outCount = indexEntry.getLength() / sizeof(T);

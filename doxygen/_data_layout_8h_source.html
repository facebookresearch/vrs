481 template <class T>
482 T* getFixedData(size_t offset, size_t size) {
483 if (mappedDataLayout_ != nullptr) {
484 return mappedDataLayout_->getFixedData<T>(offset, size);
485 }
486 if (offset != kNotFound && offset + size <= fixedData_.size()) {
487 return reinterpret_cast<T*>(fixedData_.data() + offset);
488 }
489 return nullptr;
490 }

504 template <class T>
505 T* getVarData(size_t varPieceIndex, size_t& outCount) {
506 if (mappedDataLayout_ != nullptr) {
507 return mappedDataLayout_->getVarData<T>(varPieceIndex, outCount);
508 }
509 if (varPieceIndex < varSizePieces_.size()) {
510 const IndexEntry& indexEntry = getVarSizeIndex()[varPieceIndex];
511 if (indexEntry.getOffset() + indexEntry.getLength() <= varData_.size()) {
512 outCount = indexEntry.getLength() / sizeof(T);
513 return reinterpret_cast<T*>(varData_.data() + indexEntry.getOffset());
514 }
515 }
516 outCount = 0;
517 return nullptr;
518 }

724template <typename T, size_t Size>
725struct DataLayoutStructArray : public vrs::DataLayoutStruct {
726 DATA_LAYOUT_STRUCT(DataLayoutStructArray)
727 std::array<T, Size> array{createArrayHelper<T>(std::make_index_sequence<Size>())};
728
729 T& operator[](const size_t index) {
730 return array[index];
731 }
732
733 constexpr const T& operator[](const size_t index) const {
734 return array[index];
735 }
736
737 constexpr std::size_t size() const noexcept {
738 return array.size();
739 }
740
741 template <typename S, size_t... Indices>
742 static constexpr auto createArrayHelper(std::index_sequence<Indices...>) {
743 return std::array<S, sizeof...(Indices)>{S{std::to_string(Indices)}...};
744 }

36template <typename T>
37class DataPieceValue : public DataPiece {
38 static_assert(
39 !std::is_same<T, bool>::value,
40 "DataPieceValue does not support bool. Use vrs::Bool instead");
41 static_assert(std::is_trivially_copyable<T>::value, "DataPieceValue only supports POD types.");

49 DataPieceValue(string label, T defaultValue)

74 T get() const {

83 if (ptr != nullptr) {

115 return true;

118 return false;
119 }

123 void setDefault(const T& defaultValue) {

146 return false;

266 unique_ptr<DataPiece> clone() const override {
267 auto other = std::make_unique<DataPieceValue<T>>(getLabel());
268 other->tags_ = tags_;
269 other->required_ = required_;
270 other->properties_ = properties_;
271 if (defaultValue_) {
272 other->defaultValue_ = std::make_unique<T>(*defaultValue_);
273 }
274 return other;
275 }

277 protected:

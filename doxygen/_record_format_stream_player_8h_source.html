167  template <class T>
168  T& getCachedLayout(
169  vector<unique_ptr<DataLayout>>& layoutCache,
170  DataLayout& layout,
171  size_t blockIndex) {
172  if (layoutCache.size() <= blockIndex) {
173  layoutCache.resize(blockIndex + 1);
174  }
175  if (!layoutCache[blockIndex]) {
176  T* expectedLayout = new T;
177  layoutCache[blockIndex].reset(expectedLayout);
178  expectedLayout->mapLayout(layout);
179  }
180  return *reinterpret_cast<T*>(layoutCache[blockIndex].get());
181  }
